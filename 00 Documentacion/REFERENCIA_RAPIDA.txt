🎯 EJEMPLOS DE USO

  -- Ver NFTs aprobados
  SELECT * FROM nft.NFT WHERE StatusCode = 'APPROVED';

  -- Listar subastas activas
  SELECT * FROM auction.Auction WHERE StatusCode = 'ACTIVE';

  -- Revisar emails pendientes
  SELECT * FROM audit.EmailOutbox WHERE StatusCode = 'PENDING';
